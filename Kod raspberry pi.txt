# Na produkcji indeks zależałby od modemu/RTSP drona. 
# W dewelopmencie używamy kamery USB (index 1) lub wbudowanej (index 0).
CAMERA_INDEX = 1 
STREAM_FPS = 25
AI_FPS = 10
POSE_CONFIDENCE = 0.2
MIN_ELAPSED_TIME_THRESHOLD = 3
INCIDENT_COOLDOWN = 12
CAMERA_LABEL = "Kamera Drona"
logging.basicConfig(level=logging.INFO, format="%(asctime)s [%(levelname)s] %(message)s")
logger = logging.getLogger("skyguard.hub")
class FallDetector:
    def __init__(self):
        self.reset()
    def reset(self):
        self.previous_y_values = None
        self.first_point_threshold = None
        self.second_point_threshold = None
        self.falling_threshold = None
        self.fallen_state = False
        self.fall_start_time = None
        self.elapsed_time_states = []
        self.fall_alerted = False
        self.initialized = False
        self.start_time = time.time()
        self._last_alert_time = 0
    def _get_y_values(self, result) -> list:
        if result.keypoints is None or len(result.keypoints.xy) == 0:
            return []
        return [float(kp[1]) for kp in result.keypoints.xy[0] if float(kp[1]) != 0]
    def initialize(self, y_first: list, y_second: list) -> bool:
        if len(y_first) < 6 or len(y_second) < 6: return False
        self.falling_threshold = ((y_first[-1] - y_first[0]) * 2 / 3) + 20
        self.first_point_threshold = abs(y_first[0] - y_second[0]) + 15
        self.second_point_threshold = abs(y_first[5] - y_second[5]) + 15
        self.initialized = True
        return True
    def process(self, result) -> tuple:
        y_values = self._get_y_values(result)
        if len(y_values) < 6:
            if self.fallen_state and self.fall_start_time:
                if time.time() - self.fall_start_time >= MIN_ELAPSED_TIME_THRESHOLD:
                    return self._trigger_alert()
            return "no_human", False
        if not self.initialized:
            if self.previous_y_values is None:
                self.previous_y_values = y_values
                return "initializing", False
            self.initialize(self.previous_y_values, y_values)
            self.previous_y_values = y_values
            return "initializing", False
        state, alert = "safe", False
        minimum, maximum = min(y_values), max(y_values)
        if self.previous_y_values and len(self.previous_y_values) >= 6:
            pt1_diff = abs(self.previous_y_values[0] - y_values[0])
            pt2_diff = abs(self.previous_y_values[5] - y_values[5])
            if self.falling_threshold and (maximum - minimum <= self.falling_threshold):
                if pt1_diff <= self.first_point_threshold and pt2_diff <= self.second_point_threshold:
                    state = "laying_down"
                    if self.fallen_state:
                        state, alert = self._check_time()
                    else:
                        self.fall_start_time = None
                else:
                    state = "fallen"
                    if self.fallen_state:
                        state, alert = self._check_time()
                    else:
                        self.fallen_state = True
                        self.fall_start_time = time.time()
            else:
                if not self.fall_alerted:
                    self.fallen_state, self.fall_start_time = False, None
                state = "safe"
        if not self.fallen_state and (time.time() - self.start_time > 5):
            self.initialize(y_values, y_values)
            self.start_time = time.time()
        self.previous_y_values = y_values
        return state, alert
    def _check_time(self) -> tuple:
        if self.fall_start_time and (time.time() - self.fall_start_time >= MIN_ELAPSED_TIME_THRESHOLD):
            return self._trigger_alert()
        return "fallen", False
    def _trigger_alert(self) -> tuple:
        now = time.time()
        if now - self._last_alert_time < INCIDENT_COOLDOWN:
            return "fallen", False
        logger.warning("Incydent: Upadek wykryty!")
        self._last_alert_time = now
        self.fall_alerted = True
        self.fallen_state = False
        self.fall_start_time = None
        return "fall_alert", True
class State:
    latest_frame, annotated_frame = None, None
    incidents, websocket_clients = [], []
    drone = {"state": "patrolling", "battery_pct": 87.5, "floor": 1}
    lock = Lock()
    model = None
    fall_detector = FallDetector()
state = State()
app = FastAPI(title="SkyGuard EDU API")
app.add_middleware(CORSMiddleware, allow_origins=["*"], allow_credentials=True, allow_methods=["*"], allow_headers=["*"])
def load_model():
    from ultralytics import YOLO
    pose_path = "yolov8n-pose.pt"
    if not os.path.exists(pose_path):
        url = "https://github.com/ultralytics/assets/releases/download/v8.4.0/yolov8n-pose.pt"
        logger.info("Pobieranie modelu YOLOv8 Pose...")
        warnings.filterwarnings("ignore")
        resp = requests.get(url, verify=False, stream=True)
        with open(pose_path, "wb") as f:
            for chunk in resp.iter_content(8192): f.write(chunk)
    state.model = YOLO(pose_path)
SKELETON = [(0,1),(0,2),(1,3),(2,4),(5,6),(5,7),(7,9),(6,8),(8,10),(5,11),(6,12),(11,12),(11,13),(13,15),(12,14),(14,16)]
def _draw_pose(frame, result):
    if result.keypoints:
        for kpts in result.keypoints.xy:
            pts = [(int(kp[0]), int(kp[1])) for kp in kpts]
            for a, b in SKELETON:
                if a < len(pts) and b < len(pts) and pts[a] != (0,0) and pts[b] != (0,0):
                    cv2.line(frame, pts[a], pts[b], (0, 180, 255), 2, cv2.LINE_AA)
    return frame
def camera_loop():
    logger.info(f"Oczekiwanie na wideo drona (port={CAMERA_INDEX})...")
    cap = cv2.VideoCapture(CAMERA_INDEX)
    cap.set(cv2.CAP_PROP_FRAME_WIDTH, 1280)
    cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 720)
    frame_count, ai_every_n = 0, max(1, int(STREAM_FPS / AI_FPS))
    
    while True:
        ret, frame = cap.read()
        if not ret: continue
        frame_count += 1
        annotated = frame.copy()
        if state.model and frame_count % ai_every_n == 0:
            results = state.model.predict(source=frame, conf=POSE_CONFIDENCE, verbose=False)
            if results:
                det_state, alert = state.fall_detector.process(results[0])
                annotated = _draw_pose(annotated, results[0])
                if alert:
                    asyncio.run_coroutine_threadsafe(
                        _emit_incident("fall", 0.95, "Upadek ucznia wykryty przez drona"), _event_loop)
        # Rysowanie OSD
        h, w = annotated.shape[:2]
        cv2.rectangle(annotated, (0,0), (w,40), (0,0,0), -1)
        cv2.putText(annotated, "SkyGuard EDU", (10,27), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (100,200,255), 2)
        cv2.putText(annotated, CAMERA_LABEL, (w-150, h-12), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (150,150,150), 1)
        with state.lock:
            state.latest_frame, state.annotated_frame = frame, annotated
async def _emit_incident(inc_type, conf, desc):
    inc = {"id": len(state.incidents)+1, "incident_type": inc_type, "confidence": conf, "location": CAMERA_LABEL, "description": desc, "created_at": datetime.now().isoformat()}
    state.incidents.insert(0, inc)
    for ws in state.websocket_clients: await ws.send_text(json.dumps({"type": "incident", "data": inc}, default=str))
@app.get("/stream/live")
async def live_stream():
    async def generator():
        while True:
            with state.lock:
                frame = state.annotated_frame if state.annotated_frame is not None else state.latest_frame
            if frame is None:
                await asyncio.sleep(0.04); continue
            _, jpg = cv2.imencode(".jpg", frame, [cv2.IMWRITE_JPEG_QUALITY, 82])
            yield (b"--frame\r\nContent-Type: image/jpeg\r\n\r\n" + jpg.tobytes() + b"\r\n")
            await asyncio.sleep(1 / STREAM_FPS)
    return StreamingResponse(generator(), media_type="multipart/x-mixed-replace; boundary=frame")
@app.get("/incidents/")
async def list_incidents(): return {"incidents": state.incidents}
@app.get("/drone/status")
async def drone_status(): return state.drone
@app.websocket("/ws")
async def websocket_endpoint(websocket: WebSocket):
    await websocket.accept(); state.websocket_clients.append(websocket)
    try:
        while True: await websocket.receive_text()
    except WebSocketDisconnect: state.websocket_clients.remove(websocket)
async def main():
    global _event_loop; _event_loop = asyncio.get_event_loop()
    load_model()
    Thread(target=camera_loop, daemon=True).start()
    config = uvicorn.Config(app=app, host="0.0.0.0", port=8000, log_level="warning")
    await uvicorn.Server(config).serve()
if __name__ == "__main__": asyncio.run(main())
